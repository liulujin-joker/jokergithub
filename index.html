🚀 快速开始

# 1. 克隆仓库 git clone https://github.com/liulujin-joker/jokergithub.git cd jokergithub/vivo-ai-imaging-agent # 2. 安装依赖 pip install -r requirements.txt # 3. 运行演示 python main.py --demo # 4. 交互对话 python main.py # 5. 单次查询 python main.py --query "帮我把照片变清晰"

vivo-ai-imaging-agent/ ├── main.py # 👈 智能体主体入口 VivoImagingAgent ├── requirements.txt ├── config/ │ └── settings.py # 模型/工具/记忆 全配置 ├── src/ │ ├── agent/ # 智能体核心 │ │ ├── planner.py # 规划器 Think-Plan-Reflect │ │ ├── executor.py # 执行器 │ │ └── intent_parser.py # 意图解析器 10种意图 │ ├── tools/ # 工具系统 │ │ ├── tool_registry.py # 插件化注册中心 │ │ └── imaging_tools.py # 10个AI影像工具 │ ├── memory/ # 记忆系统 │ │ └── memory_manager.py # 短期+长期双记忆 │ ├── models/schemas.py # 数据模型 │ └── utils/ # 日志/图像工具 └── docs/DESIGN.md # 完整设计文档
